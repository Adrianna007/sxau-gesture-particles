<script>
    // ==========================================
    // 1. 文字像素采样逻辑 (核心)
    // ==========================================
    function getTextCoordinates(text) {
        const canvas = document.createElement('canvas');
        const ctx = canvas.getContext('2d');
        canvas.width = 800;
        canvas.height = 400;

        ctx.fillStyle = '#000000';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        
        ctx.font = 'bold 80px "Microsoft YaHei", sans-serif';
        ctx.fillStyle = '#ffffff';
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText(text, canvas.width / 2, canvas.height / 2);

        const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height).data;
        const coordinates = [];

        // 步长决定粒子密度，步长越小粒子越多
        const step = 4; 
        for (let y = 0; y < canvas.height; y += step) {
            for (let x = 0; x < canvas.width; x += step) {
                const index = (x + y * canvas.width) * 4;
                if (imgData[index] > 128) { // 如果像素是亮色
                    // 转换为 Three.js 坐标系，居中对齐
                    coordinates.push(new THREE.Vector3(
                        (x - canvas.width / 2) * 0.1,
                        -(y - canvas.height / 2) * 0.1,
                        (Math.random() - 0.5) * 5 // Z轴微微错开增加立体感
                    ));
                }
            }
        }
        return coordinates;
    }

    // ==========================================
    // 2. Three.js 场景初始化
    // ==========================================
    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
    camera.position.z = 40;

    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    // 获取三种状态的目标坐标
    const coordsText1 = getTextCoordinates("欢迎报考");
    const coordsText2 = getTextCoordinates("山西农业大学");
    
    // 确定最大粒子数 (取两个文字中较多的那个)
    const particleCount = Math.max(coordsText1.length, coordsText2.length);

    const geometry = new THREE.BufferGeometry();
    const positions = new Float32Array(particleCount * 3);
    const targetPositions = new Float32Array(particleCount * 3);
    
    // 初始化为随机散开的球体
    for (let i = 0; i < particleCount; i++) {
        const r = 50 * Math.random();
        const theta = 2 * Math.PI * Math.random();
        const phi = Math.acos(2 * Math.random() - 1);
        
        positions[i * 3] = r * Math.sin(phi) * Math.cos(theta);
        positions[i * 3 + 1] = r * Math.sin(phi) * Math.sin(theta);
        positions[i * 3 + 2] = r * Math.cos(phi);
        
        // 初始目标设为随机
        targetPositions[i * 3] = positions[i * 3];
        targetPositions[i * 3 + 1] = positions[i * 3 + 1];
        targetPositions[i * 3 + 2] = positions[i * 3 + 2];
    }

    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    
    // 使用带有颜色的粒子材质 (山西农大通常可以配以生机勃勃的绿色/青色调)
    const material = new THREE.PointsMaterial({ 
        size: 0.3, 
        color: 0x4CAF50,
        transparent: true,
        opacity: 0.8,
        blending: THREE.AdditiveBlending
    });
    const particles = new THREE.Points(geometry, material);
    scene.add(particles);

    // ==========================================
    // 3. 状态管理与手势逻辑
    // ==========================================
    let currentTargetCoords = null; // 当前要组成的文字
    let dispersionFactor = 1.0;     // 扩散系数 (受手势张合影响)

    // 切换粒子目标的函数
    function changeTarget(coordsArray) {
        for (let i = 0; i < particleCount; i++) {
            const index = i % coordsArray.length; // 如果粒子多，循环利用坐标
            targetPositions[i * 3] = coordsArray[index].x;
            targetPositions[i * 3 + 1] = coordsArray[index].y;
            targetPositions[i * 3 + 2] = coordsArray[index].z;
        }
    }

    // ==========================================
    // 4. MediaPipe 手势识别接入
    // ==========================================
    const videoElement = document.querySelector('.input_video');
    const statusText = document.getElementById('status-text');
    const gestureText = document.getElementById('gesture-text');

    function onResults(results) {
        if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
            const landmarks = results.multiHandLandmarks[0];
            
            // 1. 计算大拇指(4)和食指(8)的距离，控制收缩与扩散
            const thumbTip = landmarks[4];
            const indexTip = landmarks[8];
            const distance = Math.sqrt(
                Math.pow(thumbTip.x - indexTip.x, 2) + 
                Math.pow(thumbTip.y - indexTip.y, 2)
            );
            // 将距离映射为扩散系数 (距离越小越收缩，距离越大越散开)
            dispersionFactor = distance * 3.0 + 0.2; 

            // 2. 简单的手势判断逻辑 (基于 Y 坐标判断手指是否伸直)
            const isIndexUp = landmarks[8].y < landmarks[6].y;
            const isMiddleUp = landmarks[12].y < landmarks[10].y;
            const isRingUp = landmarks[16].y < landmarks[14].y;

            if (isIndexUp && !isMiddleUp && !isRingUp) {
                gestureText.innerText = "1 (欢迎报考)";
                if (currentTargetCoords !== coordsText1) {
                    currentTargetCoords = coordsText1;
                    changeTarget(coordsText1);
                }
            } else if (isIndexUp && isMiddleUp && !isRingUp) {
                gestureText.innerText = "2 (山西农业大学)";
                if (currentTargetCoords !== coordsText2) {
                    currentTargetCoords = coordsText2;
                    changeTarget(coordsText2);
                    // 切换为农大绿
                    material.color.setHex(0x2E8B57); 
                }
            } else {
                gestureText.innerText = "张合控制中...";
            }
        }
    }

    const hands = new Hands({locateFile: (file) => {
        return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;
    }});
    hands.setOptions({
        maxNumHands: 1,
        modelComplexity: 1,
        minDetectionConfidence: 0.7,
        minTrackingConfidence: 0.7
    });
    hands.onResults(onResults);

    const cameraUtils = new Camera(videoElement, {
        onFrame: async () => {
            await hands.send({image: videoElement});
        },
        width: 640,
        height: 480
    });
    cameraUtils.start().then(() => {
        statusText.innerText = "系统状态：摄像头已就绪";
        statusText.style.color = "#4CAF50";
    });

    // ==========================================
    // 5. 动画渲染循环 (缓动插值)
    // ==========================================
    function animate() {
        requestAnimationFrame(animate);

        const posAttr = geometry.attributes.position;
        const currentArr = posAttr.array;

        for (let i = 0; i < particleCount; i++) {
            const idxX = i * 3;
            const idxY = i * 3 + 1;
            const idxZ = i * 3 + 2;

            // 目标位置叠加扩散扰动 (dispersionFactor)
            const targetX = targetPositions[idxX] * dispersionFactor;
            const targetY = targetPositions[idxY] * dispersionFactor;
            const targetZ = targetPositions[idxZ] * dispersionFactor;

            // 线性插值 (Lerp)，0.05 决定了粒子飞行的速度，数值越大飞得越快
            currentArr[idxX] += (targetX - currentArr[idxX]) * 0.05;
            currentArr[idxY] += (targetY - currentArr[idxY]) * 0.05;
            currentArr[idxZ] += (targetZ - currentArr[idxZ]) * 0.05;
        }

        posAttr.needsUpdate = true;
        
        // 场景缓慢旋转，增加高级感
        particles.rotation.y += 0.002;
        particles.rotation.x = Math.sin(Date.now() * 0.001) * 0.1;

        renderer.render(scene, camera);
    }

    animate();

    // 监听窗口缩放
    window.addEventListener('resize', () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
    });
</script>
